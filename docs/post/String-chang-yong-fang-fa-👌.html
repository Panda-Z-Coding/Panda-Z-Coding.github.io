字符序列加密算法

通过password这个String对象的字符序列作为密码存放在数组里对另一个sourceString的字符序列进行加密！

char[] p = password.toCharArray(); //设p的长度为n

以下是具体实现

public class EncryptAndDecrypt {
    String encrypt(String sourceString,String password) {
        char[] p = password.toCharArray ();
        int n = p.length;
        char[] c = sourceString.toCharArray ();
        int m = c.length;
        for(int k=0;k<m;k++) {
            int mima = c[k] + p[k % n];
            c[k] = (char) mima;
        }
        return new String (c);
    }
    String decrypt(String sourceString, String password) {
        char[] p = password.toCharArray ();
        int n = p.length;
        char[] c = sourceString.toCharArray ();
        int m = c.length;
        for (int k = 0; k < m; k++) {
            int mima = c[k] - p[k % n];
            c[k] = (char) mima;
        }
        return new String (c);
    }
}
